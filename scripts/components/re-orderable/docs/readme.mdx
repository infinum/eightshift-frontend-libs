# ReOrderable

A simple list that allows drag&drop re-ordering of items.

## Usage

To use it:
1. Import `ReOrderable` and `ReOrderableItem` from Frontend Libs
2. Add the `ReOrderable` into the options of your block/component

```jsx
import { LinkEditComponent } from '@eightshift/frontend-libs/scripts/components';
import { getAttrKey } from '@eightshift/frontend-libs/scripts/helpers';

// ...

export const VideoOptions = (attributes) =>  {
	const {
		setAttributes,
	} = attributes;

	const videoCaptions = checkAttr('videoCaptions', attributes, manifest);

	return (
			// ...

			<ReOrderable
				icon={icons.videoSubtitleAlt}
				label={__('Captions', 'eightshift-frontend-libs')}
				items={videoCaptions}
				attributeName={getAttrKey('videoCaptions', attributes, manifest)}
				setAttributes={setAttributes}
			>
				{videoCaptions.map((item) => (
					<ReOrderableItem
						key={item.id} // id is auto-generated
						icon={item.icon} // you'll probably have a function to return an icon based on type
						title={item.title}
					>
						// Your options here
					</ReOrderableItem>
				))}
			</ReOrderable>
	)
}
```
